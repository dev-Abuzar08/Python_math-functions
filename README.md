def even_odd(num):
    if num % 2 == 0:
        return "Even"
    else:
         return "Odd"
print("Examples:")

print(even_odd(5))
print(even_odd(8))
print(even_odd(24))



def power(base, exponent):
    return base ** exponent
print("Examples:")

print(power(2,3))
print(power(10,50))
print(power(2,8))


def power(x, y):
    return x ** y
